1. Menampilkan daftar sewaan yang menyewa paket
SELECT pesanan.id_pesanan, pemesan.nama_pemesan, pesanan.tgl_mulai, pesanan.tgl_selesai,  paket.nama_paket 
FROM pesanan
JOIN pemesan ON pemesan.id_pemesan = pesanan.id_pemesan 
JOIN paket_pesanan ON paket_pesanan.id_pesanan = pesanan.id_pesanan 
JOIN paket ON paket.id_paket = paket_pesanan.id_paket

2. Menampilkan daftar sewaan yang menyewa beberapa barang (non-paket)
SELECT pesanan.id_pesanan, pemesan.nama_pemesan, pesanan.tgl_mulai, pesanan.tgl_selesai, barang_pesanan.jumlah_barang, barang.nama_barang 
FROM pesanan
JOIN pemesan ON pemesan.id_pemesan = pesanan.id_pemesan 
JOIN barang_pesanan ON barang_pesanan.id_pesanan = pesanan.id_pesanan 
JOIN barang ON barang.id_barang = barang_pesanan.id_barang

4. Menampilkan detail sewaan yang paket beserta isi paket
SELECT pesanan.id_pesanan, pemesan.nama_pemesan, pesanan.tgl_mulai, pesanan.tgl_selesai, paket.nama_paket, isi_paket.jumlah_barang, barang.nama_barang FROM pesanan JOIN pemesan ON pemesan.id_pemesan = pesanan.id_pemesan JOIN paket_pesanan ON paket_pesanan.id_pesanan = pesanan.id_pesanan JOIN paket ON paket.id_paket = paket_pesanan.id_paket JOIN isi_paket ON isi_paket.id_paket = paket.id_paket JOIN barang ON barang.id_barang = isi_paket.id_barang

5. Menampilkan detail penyewa dari suatu sewaan
